size (200, 50)

states {
    (alive, 255, 255, 255),
    (infected, 255, 0, 0, quantity 1),
    (resistant, 0, 255, 0, proportion 0.6),
    (dead, 0, 0, 0, quantity 0),
}

transitions {
    (alive, infected, infected >= 1),
    (infected, dead, rand 0.5, delay 5),
    (infected, resistant, true, delay 3),
}
